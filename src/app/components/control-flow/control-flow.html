} @else {

@if (userRole === 'admin') {

👤 Invité

2. @for avec track et @empty

Liste de fruits :

    @for (fruitItem of fruits; track fruitItem){
  • {{fruitItem}}
    • }

Liste d'utilisateurs (avec index) :

    @for (userItem of users; track $index) {
  • {{$index + 1}}. {{userItem.name}}
    • }

Liste vide (avec @empty) :

    @for (item of emptyList; track item) {
  • {{ item }}
    • } @empty {
  • Aucun élément à afficher.
    • }

} @case ('error'){

Une erreur est survenue.

} @default {

État inconnu.

} }
